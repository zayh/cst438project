DROP TABLE album;
CREATE TABLE IF NOT EXISTS album (
	album_id BIGINT NOT NULL AUTO_INCREMENT UNIQUE PRIMARY KEY, 
	release_date DATE;
	album_name VARCHAR (100) NOT NULL,
	genre VARCHAR(50) NULL,
	url_to_buy VARCHAR(100) NULL,
	band_id BIGINT,
	FOREIGN KEY (band_id) 
		REFERENCES band(band_id) 
		ON DELETE CASCADE
);
