DROP TABLE map_song_to_album;
CREATE TABLE IF NOT EXISTS map_song_to_album (
	map_song_to_album_id BIGINT NOT NULL AUTO_INCREMENT UNIQUE PRIMARY KEY,  
	track_number INT, 
	album_id BIGINT,
	song_id BIGINT,
	FOREIGN KEY (album_id)
		REFERENCES album(album_id)
		ON DELETE CASCADE,
	FOREIGN KEY (song_id) 
		REFERENCES song(song_id) 
		ON DELETE CASCADE
);
