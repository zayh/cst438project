DROP TABLE rating;
CREATE TABLE IF NOT EXISTS rating (
	rating_id BIGINT NOT NULL AUTO_INCREMENT UNIQUE PRIMARY KEY,  
	account_id BIGINT,
	album_id BIGINT,
	rating INT,
	comment TEXT,
	FOREIGN KEY (album_id)
		REFERENCES album(album_id)
		ON DELETE CASCADE,
	FOREIGN KEY (account_id) 
		REFERENCES account(account_id) 
		ON DELETE CASCADE
);
